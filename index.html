function updateChart() { const ctx = document.getElementById('bpChart').getContext('2d'); if(myChart) myChart.destroy(); // ดึงข้อมูลวันที่ล่าสุด 7 วัน (ถ้าไม่มีให้สร้างแกนวันปัจจุบันรอไว้) let labels = [...new Set([...db.hospital, ...db.home_m, ...db.home_e].map(x => x.date))].slice(-7); if (labels.length === 0) labels = [new Date().toLocaleDateString('th-TH')]; const getDailyAvg = (dataArr, labelDate) => { const dayData = dataArr.filter(x => x.date === labelDate); return dayData.length ? dayData.reduce((a,b)=>a+b.s,0)/dayData.length : null; }; myChart = new Chart(ctx, { type: 'line', data: { labels: labels, datasets: [ { label: 'รพ.', data: labels.map(l => getDailyAvg(db.hospital, l)), borderColor: '#ef4444', backgroundColor: '#ef4444', pointStyle: 'rectRot', radius: 6, spanGaps: true // เชื่อมเส้นแม้ข้อมูลขาดช่วง }, { label: 'เฉลี่ยบ้าน-เช้า', data: labels.map(l => getDailyAvg(db.home_m, l)), borderColor: '#4ade80', // สีเขียวสว่าง backgroundColor: '#4ade80', tension: 0.3, pointRadius: 4, spanGaps: true }, { label: 'เฉลี่ยบ้าน-เย็น', data: labels.map(l => getDailyAvg(db.home_e, l)), borderColor: '#facc15', // สีเหลือง backgroundColor: '#facc15', tension: 0.3, pointRadius: 4, spanGaps: true }, { label: 'เป้าหมาย', data: Array(labels.length).fill(135), borderColor: '#ffffff', // บังคับเป็นสีขาว borderWidth: 2, borderDash: [5, 5], pointRadius: 0, fill: false, z: -1 // ให้เส้นเป้าหมายอยู่ด้านหลังจุดข้อมูล } ] }, options: { responsive: true, maintainAspectRatio: false, layout: { padding: { top: 10 } }, scales: { y: { min: 90, max: 200, ticks: { stepSize: 10, color: '#94a3b8', font: { size: 9 } }, grid: { color: 'rgba(255,255,255,0.05)' } }, x: { ticks: { color: '#94a3b8', font: { size: 8 } }, grid: { display: false } } }, plugins: { legend: { display: false } // เราใช้ Legend ที่สร้างเองด้านล่างกราฟแทน } } }); }
